Q3. Write a C++ program to create a class called "Rectangle" and calculate area and perimeter.

using namespace std;

class Rectangle {
    int length, width;

public:
    void setValues(int l, int w) {
        length = l;
        width = w;
    }

    int area() {
        return length * width;
    }

    int perimeter() {
        return 2 * (length + width);

    void display() {
        cout << "Area: " << area() << endl;
        cout << "Perimeter: " << perimeter() << endl;
    }
};

int main() {
    Rectangle r;
    r.setValues(10, 5);
    r.display();
}

OUTPUT

Area: 50
Perimeter: 30

Q7. Write a C++ program to create a "Bank" and "Account" class for transactions.

#include <iostream>
#include <string>
using namespace std;

class Account {
    string name;
    int accNo;
    float balance;

public:
    void setDetails(string n, int a, float b) {
        name = n;
        accNo = a;
        balance = b;
    }

    void deposit(float amt) {
        balance += amt;
        cout << "Deposited: " << amt << endl;
    }

    void withdraw(float amt) {
        if(amt <= balance) {
            balance -= amt;
            cout << "Withdrawn: " << amt << endl;
        } else {
            cout << "Insufficient Balance\n";
        }
    }

    void display() {
        cout << "Name: " << name << endl;
        cout << "Account No: " << accNo << endl;
        cout << "Balance: " << balance << endl;
    }
};

class Bank {
public:
    void performTransaction(Account &a) {
        a.deposit(1000);
        a.withdraw(500);
    }
};

int main() {
    Account a1;
    Bank b;

    a1.setDetails("Vansh", 101, 5000);

    cout << "Before Transaction:\n";
    a1.display();

    b.performTransaction(a1);

    cout << "\nAfter Transaction:\n";
    a1.display();
}

OUTPUT

Before Transaction:
Name: Vansh
Account No: 101
Balance: 5000
Deposited: 1000
Withdrawn: 500

After Transaction:
Name: Vansh
Account No: 101
Balance: 5500

Q16. Write a C++ program using abstract "Shape" class and derived classes.

#include <iostream>
using namespace std;

class Shape {
public:
    virtual void area() = 0;
};

class Circle : public Shape {
    float r;
public:
    Circle(float radius) { r = radius; }

    void area() {
        cout << "Area of Circle: " << 3.14 * r * r << endl;
    }
};

class Rectangle : public Shape {
    float l, b;
public:
    Rectangle(float length, float breadth) {
        l = length;
        b = breadth;
    }

    void area() {
        cout << "Area of Rectangle: " << l * b << endl;
    }
};

int main() {
    Shape *s;
    Circle c(5);
    Rectangle r(4, 6);

    s = &c;
    s->area();

    s = &r;
    s->area();
}

OUTPUT

Area of Circle: 78.5
Area of Rectangle: 24

Q18. Write a C++ program to create a "Restaurant" class with ratings.

#include <string>
using namespace std;

class Restaurant {
    string name;
    int ratings[5];
    int count;

public:
    Restaurant(string n) {
        name = n;
        count = 0;
    }

    void addRating(int r) {
        if (count < 5) ratings[count++] = r;
    }

    void displayRatings() {
        cout << "Restaurant: " << name << endl;
        cout << "Ratings:\n";
        for (int i = 0; i < count; i++)
            cout << i + 1 << ". " << ratings[i] << endl;
    }

    void averageRating() {
        int sum = 0;
        for (int i = 0; i < count; i++) sum += ratings[i];
        cout << "Average Rating: " << (float)sum / count << endl;
    }
};

int main() {
    Restaurant r("Dominos");

    r.addRating(5);
    r.addRating(4);
    r.addRating(3);

    r.displayRatings();
    r.averageRating();
}

OUTPUT

Restaurant: Dominos
Ratings:
1. 5
2. 4
3. 3
