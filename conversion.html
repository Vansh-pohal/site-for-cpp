Q5. C++ Program to convert String to float

#include <iostream>
using namespace std;

int main(){
    char s[100];
    cin>>s;

    float num = 0, frac = 0, div = 1;
    int i = 0, flag = 0;

    while(s[i] != '\0'){
        if(s[i] == '.'){
            flag = 1;
        }
        else if(flag == 0){
            num = num * 10 + (s[i] - '0');
        }
        else{
            frac = frac * 10 + (s[i] - '0');
            div *= 10;
        }
        i++;
    }

    cout<<(num + frac/div);
}

OUTPUT

Input: 12.34
Output: 12.34

Q6. C++ Program to convert float to String

#include <iostream>
using namespace std;

int main(){
    float n;
    cin>>n;

    int intPart = (int)n;
    float frac = n - intPart;

    char s[100];
    int i = 0;

    while(intPart > 0){
        s[i++] = (intPart % 10) + '0';
        intPart /= 10;
    }

    for(int j=i-1; j>=0; j--) cout<<s[j];

    cout<<".";

    for(int k=0; k<3; k++){
        frac *= 10;
        int digit = (int)frac;
        cout<<digit;
        frac -= digit;
    }
}

OUTPUT

Input: 12.34
Output: 12.340

Q11. C++ Program to convert Binary to Decimal.

#include <iostream>
using namespace std;

int main() {
    int bin, ld, pow = 1, res = 0;

    cout<<"ENTER BINARY: ";
    cin>>bin;

    while(bin > 0){
        ld = bin % 10;
        res += ld * pow;
        pow *= 2;
        bin = bin / 10;
    }

    cout<<"DECIMAL : "<<res;
}

OUTPUT

ENTER BINARY: 1011
DECIMAL : 11

Q13. C++ Program to convert Hexadecimal to Decimal.

#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char hex[25];
    int res = 0, power = 1;

    cout<<"ENTER HEX: ";
    cin>>hex;

    int len = strlen(hex);

    for(int i = len - 1; i >= 0; i--){
        if(hex[i] >= '0' && hex[i] <= '9')
            res += (hex[i] - '0') * power;
        else if(hex[i] >= 'A' && hex[i] <= 'F')
            res += (hex[i] - 'A' + 10) * power;
        else if(hex[i] >= 'a' && hex[i] <= 'f')
            res += (hex[i] - 'a' + 10) * power;

        power *= 16;
    }

    cout<<"DECIMAL : "<<res;
}

OUTPUT

ENTER HEX: A
DECIMAL : 10

Q14. C++ Program to convert Decimal to Hexadecimal.

#include <iostream>
using namespace std;

int main() {
    int n, rem, i = 0;
    char hex[25];

    cout<<"ENTER DECIMAL: ";
    cin>>n;

    while(n > 0){
        rem = n % 16;

        if(rem < 10)
            hex[i] = rem + '0';
        else
            hex[i] = rem - 10 + 'A';

        i++;
        n = n / 16;
    }

    cout<<"HEXADECIMAL : ";
    for(int j = i - 1; j >= 0; j--){
        cout<<hex[j];
    }
}

OUTPUT

ENTER DECIMAL: 26
HEXADECIMAL : 1A
