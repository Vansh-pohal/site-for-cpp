Q1. Write a C++ program to find the smallest number among three numbers.

    #include <iostream>
    using namespace std;

    int smallest(int a, int b, int c){
        if(a <= b && a <= c)
            return a;
        else if(b <= a && b <= c)
            return b;
        else
            return c;
    }

    int main(){
        int a, b, c;
        cout << "Enter three numbers: ";
        cin >> a >> b >> c;

        cout << "Smallest number is: " << smallest(a, b, c);
    }

OUTPUT

    Enter three numbers: 10 5 8
    Smallest number is: 5

Q3. Write a C++ program to display the middle character of a string.

    #include <iostream>
    #include <cstring>
    using namespace std;

    // Function to find middle character(s)
    void findMiddle(char str[]) {
        int len = strlen(str);

        if (len % 2 == 0) {
            cout << "Middle characters are: " << str[len/2 - 1] << " and " << str[len/2];
        } else {
            cout << "Middle character is: " << str[len/2];
        }
    }

    int main() {
        char str[100];

        cout << "Enter a string: ";
        cin.getline(str, 100);

        findMiddle(str);   
        return 0;
    }

OUTPUT

    Enter a string: hello
    Middle character is: l

Q4. Write a C++ program to find all twin prime numbers less than 100.

    #include <iostream>
    using namespace std;

    int is_prime(int n){
        if(n < 2) return 0;
        for(int i = 2; i < n/2; i++){
            if(n % i == 0)
                return 0;
        }
        return 1;
    }
    
    int main(){
        cout << "Twin primes less than 100:\n";

        for (int i = 2; i < 100; i++) {
            if (is_prime(i) && is_prime(i + 2)) {
                cout << "(" << i << ", " << i + 2 << ")" << endl;
            }
        }

        return 0;
    }

OUTPUT

    Twin primes less than 100:
    (3, 5)
    (5, 7)
    (11, 13)
    (17, 19)
    (29, 31)
    (41, 43)
    (59, 61)
    (71, 73)

Q5. Write a C++ program to count the number of digits with value 2 in a non-negative integer.

    #include <iostream>
    using namespace std;

    int countTwos(int n) {
        int count = 0;
        while (n > 0) {
            if (n % 10 == 2)
                count++;
            n /= 10;
        }
        return count;
    }

    int main() {
        int num;
        cout << "Enter a number: ";
        cin >> num;

        cout << "Count of digit 2: " << countTwos(num);
        return 0;
    }

OUTPUT

    Enter a number: 22452
    Count of digit 2: 3

Q6. Write a C++ program that accepts three integers and checks whether they are consecutive or not.

    #include <iostream>
    using namespace std;

    bool isConsecutive(int a, int b, int c) {
        if ((a + 1 == b && b + 1 == c) || 
            (a - 1 == b && b - 1 == c) ||
            (b + 1 == a && a + 1 == c) ||
            (b - 1 == a && a - 1 == c) ||
            (c + 1 == a && a + 1 == b) ||
            (c - 1 == a && a - 1 == b))
            return true;
        return false;
    }

    int main() {
        int a, b, c;
        cout << "Enter three numbers: ";
        cin >> a >> b >> c;

        cout << isConsecutive(a, b, c);
        return 0;
    }

OUTPUT

    Enter three numbers: 5 6 7
    1

Q7. Write a C++ program to check whether all characters in a string are vowels or not.

    #include <iostream>
    #include <cctype>
    using namespace std;

    bool allVowels(char str[]) {
        for (int i = 0; str[i] != '\0'; i++) {
            char ch = tolower(str[i]);
            if (!(ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u'))
                return false;
        }
        return true;
    }

    int main() {
        char str[100];

        cout << "Enter a string: ";
        cin.getline(str, 100);

        cout << allVowels(str);
        return 0;
    }

OUTPUT

    Enter a string: aeiou
    1
