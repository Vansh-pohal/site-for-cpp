Q5. Create BankAccount and override withdraw() in SavingsAccount.

#include <iostream>
using namespace std;

class BankAccount {
protected:
    int balance;

public:
    BankAccount(int b) {
        balance = b;
    }

    void deposit(int amt) {
        balance += amt;
    }

    virtual void withdraw(int amt) {
        if (amt > balance)
            cout << "Insufficient balance\n";
        else {
            balance -= amt;
            cout << "Withdraw successful\n";
        }
    }

    int getBalance() {
        return balance;
    }
};

class SavingsAccount : public BankAccount {
public:
    SavingsAccount(int b) : BankAccount(b) {}

    void withdraw(int amt) {
        if (balance - amt < 100)
            cout << "Minimum balance required\n";
        else {
            balance -= amt;
            cout << "Withdraw successful\n";
        }
    }
};

int main() {
    SavingsAccount acc(500);
    acc.deposit(200);
    acc.withdraw(300);
    cout << acc.getBalance();
}

OUTPUT

Withdraw successful
400

Q7. Create Person and override getLastName() in Employee.

#include <iostream>
using namespace std;

class Person {
protected:
    string first, last;

public:
    Person(string f, string l) {
        first = f;
        last = l;
    }

    string getFirstName() {
        return first;
    }

    virtual string getLastName() {
        return last;
    }
};

class Employee : public Person {
    int id;
    string title;

public:
    Employee(string f, string l, int i, string t)
        : Person(f, l) {
        id = i;
        title = t;
    }

    string getLastName() {
        return last + " (" + title + ")";
    }
};

int main() {
    Employee e("Sarthak", "Gussain", 101, "Developer");
    cout << e.getFirstName() << endl;
    cout << e.getLastName();
}

OUTPUT

Sarthak
Gussain (Developer)

Q8. Create Shape and override getArea() and getPerimeter() in Circle.

#include <iostream>
using namespace std;

class Shape {
public:
    virtual int getArea() { return 0; }
    virtual int getPerimeter() { return 0; }
};

class Circle : public Shape {
    int r;

public:
    Circle(int r) {
        this->r = r;
    }

    int getArea() {
        return (22 * r * r) / 7;
    }

    int getPerimeter() {
        return (2 * 22 * r) / 7;
    }
};

int main() {
    Circle c(7);
    cout << "Area: " << c.getArea() << endl;
    cout << "Perimeter: " << c.getPerimeter();
}

OUTPUT

Area: 154
Perimeter: 44

Q9. Create Vehicle hierarchy (Truck, Car, Motorcycle) with efficiency, distance, speed.

#include <iostream>
using namespace std;

class Vehicle {
protected:
    string make, model, fuel;
    int year;

public:
    Vehicle(string mk, string md, int y, string f) {
        make = mk; model = md; year = y; fuel = f;
    }

    virtual int fuelEfficiency() { return 0; }
    virtual int maxSpeed() { return 0; }

    int distance(int fuelUsed) {
        return fuelEfficiency() * fuelUsed;
    }
};

class Truck : public Vehicle {
public:
    Truck(string mk, string md, int y, string f)
        : Vehicle(mk, md, y, f) {}

    int fuelEfficiency() { return 8; }
    int maxSpeed() { return 120; }
};

class Car : public Vehicle {
public:
    Car(string mk, string md, int y, string f)
        : Vehicle(mk, md, y, f) {}

    int fuelEfficiency() { return 15; }
    int maxSpeed() { return 180; }
};

class Motorcycle : public Vehicle {
public:
    Motorcycle(string mk, string md, int y, string f)
        : Vehicle(mk, md, y, f) {}

    int fuelEfficiency() { return 35; }
    int maxSpeed() { return 140; }
};

int main() {
    Truck t("Tata","Prima",2022,"Diesel");
    Car c("Honda","City",2023,"Petrol");
    Motorcycle m("Yamaha","R15",2024,"Petrol");

    int fuel = 5;

    cout << "Truck distance: " << t.distance(fuel) << endl;
    cout << "Car distance: " << c.distance(fuel) << endl;
    cout << "Bike distance: " << m.distance(fuel);
}

OUTPUT

Truck distance: 40
Car distance: 75
Bike distance: 175

Q10. Create Employee hierarchy (Manager, Developer, Programmer) with bonus and reports.

#include <iostream>
using namespace std;

class Employee {
protected:
    string name;
    int salary;

public:
    Employee(string n, int s) {
        name = n;
        salary = s;
    }

    virtual int bonus() {
        return salary * 0.1;
    }

    virtual void report() {
        cout << name << " general report\n";
    }
};

class Manager : public Employee {
public:
    Manager(string n, int s) : Employee(n,s) {}

    int bonus() { return salary * 0.2; }

    void report() {
        cout << name << " manages team\n";
    }
};

class Developer : public Employee {
public:
    Developer(string n, int s) : Employee(n,s) {}

    int bonus() { return salary * 0.15; }

    void report() {
        cout << name << " writes code\n";
    }
};

class Programmer : public Employee {
public:
    Programmer(string n, int s) : Employee(n,s) {}

    int bonus() { return salary * 0.12; }

    void report() {
        cout << name << " tests programs\n";
    }
};

int main() {
    Manager m("Rahul",80000);
    Developer d("Aman",60000);
    Programmer p("Neha",50000);

    m.report();
    cout << m.bonus() << endl;

    d.report();
    cout << d.bonus() << endl;

    p.report();
    cout << p.bonus();
}

OUTPUT

Rahul manages team
16000
Aman writes code
9000
Neha tests programs
6000
