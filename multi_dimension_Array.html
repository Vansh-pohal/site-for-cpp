MULTI DIMENSION ARRAYS

Q1. Program to enter a matrix of user-defined rows and columns and print it in matrix format.

#include <iostream>
using namespace std;

int main(){
    int x,y;
    cout<<"Size of Multi Dimension Array (x axis | y axis) :";
    cin>>x>>y;

    int arr[x][y];

    cout<<"ENTER ELEMENT OF ARRAY : ";
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cin>>arr[i][j];
        }
    }

    cout<<"ELEMENT OF ARRAY IN MATRIX FORMAT :\n";
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cout<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }
}

OUTPUT

Size of Multi Dimension Array (x axis | y axis) : 2 3
ENTER ELEMENT OF ARRAY : 1 2 3 4 5 6
ELEMENT OF ARRAY IN MATRIX FORMAT :
1  2  3
4  5  6

Q2. Program to add two matrices.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{{1,2,3},{4,5,6},{7,8,9}};
    int arr_2[3][3]{{10,11,12},{13,14,15},{16,17,18}};
    int sum[3][3];

    cout<<"Elements of matrix 1\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }

    cout<<endl<<endl;

    cout<<"Elements of matrix 2 \n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr_2[i][j]<<"  ";
            sum[i][j]=arr[i][j]+arr_2[i][j];
        }
        cout<<endl;
    }

    cout<<"Sum of Both Matrix :\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<sum[i][j]<<"  ";
        }
        cout<<endl;
    }
}

OUTPUT

    4  5  6
    7  8  9

Elements of matrix 2
    10  11  12
    13  14  15
    16  17  18

Sum of Both Matrix :
    11  13  15

    23  25  27

Q3. Program to multiply two matrices.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{{1,2,3},{4,5,6},{7,8,9}};
    int arr_2[3][3]{{10,11,12},{13,14,15},{16,17,18}};
    int multiply[3][3] = {0};

    cout<<"Elements of matrix 1\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }

    cout<<endl<<endl;

    cout<<"Elements of matrix 2\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr_2[i][j]<<"  ";
        }
        cout<<endl;
    }

    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            for(int k=0;k<3;k++){
                multiply[i][j] += arr[i][k] * arr_2[k][j];
            }
        }
    }

    cout<<"Multiply of Both Matrix :\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<multiply[i][j]<<"  ";
        }
        cout<<endl;
    }
}

Q5. Program to determine whether two matrices are equal.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{0};
    int arr_2[3][3]{0};
    int flag = true;

    cout<<"Enter elements of matrix 1 ";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cin>>arr[i][j];
        }
    }

    cout<<"Enter elements of matrix 2 : ";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cin>>arr_2[i][j];
        }
    }

    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            if(arr[i][j] != arr_2[i][j]){
                flag = false;
                break;
            }
        }
        if(!flag){
            break;
        }
    }

    if(flag) cout<<"BOTH ARE EQUAL";
    else cout<<"NOT EQUAL";
}

Q6. Program to display the lower triangular matrix.

        for(int j=0;j<3;j++){
            cout<<arr[i][j]<<" ";
        }
        cout<<endl;
    }

    cout<<"Lower Triangular Matrix :\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            if(i >= j)
                cout<<arr[i][j]<<" ";
            else
                cout<<"0 ";
        }
        cout<<endl;
    }
}

Q8. Program to find the frequency of odd and even numbers in a matrix.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{{1,2,3},{4,5,6},{7,8,9}};
    int odd_count = 0, even_count = 0;

    cout<<"Element Of Matrix :\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<arr[i][j]<<" ";
            if(arr[i][j] % 2 == 0)
                even_count++;
            else
                odd_count++;
        }
        cout<<endl;
    }

    cout<<"FREQUENCY OF ODD : "<<odd_count<<endl;
    cout<<"FREQUENCY OF EVEN : "<<even_count;
}

OUTPUT

Element Of Matrix :
1 2 3
4 5 6
7 8 9

FREQUENCY OF ODD : 5
FREQUENCY OF EVEN : 4

Q9. Program to find the sum of each row and each column of a matrix.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{{1,2,3},{4,5,6},{7,8,9}};

    cout<<"Elements of matrix\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }

    // Row Sum
    for(int i=0;i<3;i++){
        int sum = 0;
        for(int j=0;j<3;j++){
            sum += arr[i][j];
        }
        cout<<"Sum Of Row "<<i+1<<" : "<<sum<<endl;
    }

    // Column Sum
    for(int j=0;j<3;j++){
        int sum = 0;
        for(int i=0;i<3;i++){
            sum += arr[i][j];
        }
        cout<<"Sum Of Column "<<j+1<<" : "<<sum<<endl;
    }
}

OUTPUT

Elements of matrix
    1  2  3
    4  5  6
    7  8  9

Sum Of Row 1 : 6
Sum Of Row 2 : 15
Sum Of Row 3 : 24

Sum Of Column 1 : 12
Sum Of Column 2 : 15
Sum Of Column 3 : 18

Q10. Program to determine whether a given matrix is an identity matrix.

#include <iostream>
using namespace std;

int main(){

    int arr[3][3]{{1,0,0},{0,1,0},{0,0,1}};

    cout<<"Matrix :\n";
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            cout<<"\t"<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }

    bool flag = true;

    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            if(i == j && arr[i][j] != 1){
                flag = false;
            }
            if(i != j && arr[i][j] != 0){
                flag = false;
            }
        }
    }

    if(flag)
        cout<<"Identity Matrix";
    else
        cout<<"Not Identity Matrix";
}

OUTPUT

Matrix :
    1  0  0
    0  1  0
    0  0  1

Identity Matrix

Q12. Program to determine whether a given matrix is a sparse matrix.

#include <iostream>
using namespace std;

int main(){

    int x,y;
    cout<<"Size of Multi Dimension Array (x axis | y axis) :";
    cin>>x>>y;

    int arr[x][y], count = 0;

    cout<<"ENTER ELEMENT OF ARRAY : ";
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cin>>arr[i][j];
            if(arr[i][j] == 0)
                count++;
        }
    }

    cout<<"Elements of matrix :\n";
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cout<<"\t"<<arr[i][j]<<"  ";
        }
        cout<<endl;
    }

    if(count > (x * y)/2)
        cout<<"It is a Sparse Matrix";
    else
        cout<<"It is not a Sparse Matrix";
}

OUTPUT

Size of Multi Dimension Array (x axis | y axis) : 3 3
ENTER ELEMENT OF ARRAY : 
1 0 0
0 0 2
0 0 0

Elements of matrix :
    1  0  0
    0  0  2
    0  0  0

It is a Sparse Matrix
