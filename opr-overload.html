OPERATOR OVERLOADING

Q4. Implement a class String to represent strings. Overload the + operator to concatenate two strings.

#include <iostream>
#include <cstring>
using namespace std;

class String {
    char str[100];

public:
    void get() {
        cin >> str;
    }

    String operator+(String s) {
        String temp;
        strcpy(temp.str, str);
        strcat(temp.str, s.str);
        return temp;
    }

    void show() {
        cout << str;
    }
};

int main() {
    String s1, s2, s3;

    s1.get();
    s2.get();

    s3 = s1 + s2;

    s3.show();
}

OUTPUT

Input: Hello World
Output: HelloWorld

Q5. Define a class Date to represent dates. Overload the < operator to compare two dates.

#include <iostream>
using namespace std;

class Date {
    int d, m, y;

public:
    void get() {
        cin >> d >> m >> y;
    }

    bool operator<(Date dt) {
        if(y < dt.y) return true;
        if(y == dt.y && m < dt.m) return true;
        if(y == dt.y && m == dt.m && d < dt.d) return true;
        return false;
    }
};

int main() {
    Date d1, d2;

    d1.get();
    d2.get();

    if(d1 < d2)
        cout << "First date is earlier";
    else
        cout << "Second date is earlier";
}

OUTPUT

Input: 1 1 2020 and 5 1 2020
Output: First date is earlier

Q6. Implement a class Money to represent money. Overload the + operator to add two amounts of money.

#include <iostream>
using namespace std;

class Money {
    int rupees, paisa;

public:
    void get() {
        cin >> rupees >> paisa;
    }

    Money operator+(Money m) {
        Money temp;
        temp.paisa = paisa + m.paisa;
        temp.rupees = rupees + m.rupees + temp.paisa / 100;
        temp.paisa = temp.paisa % 100;
        return temp;
    }

    void show() {
        cout << rupees << " Rs " << paisa << " paisa";
    }
};

int main() {
    Money m1, m2, m3;

    m1.get();
    m2.get();

    m3 = m1 + m2;

    m3.show();
}

OUTPUT

Input: 10 50 and 5 75
Output: 16 Rs 25 paisa

Q7. Create a class Rectangle to represent rectangles. Overload the == operator to check if two rectangles are equal.

#include <iostream>
using namespace std;

class Rectangle {
    int l, b;

public:
    void get() {
        cin >> l >> b;
    }

    bool operator==(Rectangle r) {
        return (l == r.l && b == r.b);
    }
};

int main() {
    Rectangle r1, r2;

    r1.get();
    r2.get();

    if(r1 == r2)
        cout << "Rectangles are equal";
    else
        cout << "Not equal";
}

OUTPUT

Input: 5 4 and 5 4
Output: Rectangles are equal

Q8. Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.

#include <iostream>
using namespace std;

class Time {
    int h, m;

public:
    void get() {
        cin >> h >> m;
    }

    friend ostream& operator<<(ostream &out, Time t) {
        if(t.h < 10) out << "0";
        out << t.h << ":";

        if(t.m < 10) out << "0";
        out << t.m;

        return out;
    }
};

int main() {
    Time t;
    t.get();

    cout << t;
}

OUTPUT

Input: 9 5
Output: 09:05
