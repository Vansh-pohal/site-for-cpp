Q7. Write a C++ program to convert decimal to binary

#include <iostream>
using namespace std;

int main(){
    int decimal, arr[64], i=0;
    cout << "Enter decimal number: ";
    cin >> decimal;

    while(decimal>0){
        arr[i++] = decimal % 2;
        decimal /= 2;
    }

    cout << "Binary: ";
    for(int j=i-1;j>=0;j--) cout << arr[j];

    return 0;
}

Q8. Write a C++ program to convert binary to decimal

#include <iostream>

    cout << "Enter binary number: ";

        n/=10;

    cout << "Decimal: " << decimal;
    return 0;
}

Q12. Count letters, spaces, numbers and other characters

#include <iostream>
#include <string>
using namespace std;

int main(){
    string text;
    int letters=0, spaces=0, numbers=0, others=0;

    cout << "Enter a string: ";
    getline(cin, text);

    for(char ch : text){
        if(isalpha(ch)) letters++;
        else if(ch==' ') spaces++;
        else if(isdigit(ch)) numbers++;
        else others++;
    }

    cout << "Letters: " << letters << endl;
    cout << "Spaces: " << spaces << endl;
    cout << "Numbers: " << numbers << endl;
    cout << "Others: " << others;

    return 0;
}

OUTPUT

Enter a string: Hello 123!
Letters: 5
Spaces: 1
Numbers: 3
Others: 1

Q20. Sum of first 100 prime numbers

#include <iostream>
using namespace std;

int main(){
    int count=0, num=2, sum=0;

    while(count<100){
        bool prime=true;

        for(int i=2;i*i<=num;i++){
            if(num%i==0){
                prime=false;
                break;
            }
        }

        if(prime){
            sum+=num;
            count++;
        }

        num++;
    }

    cout << sum;
    return 0;
}

OUTPUT

24133

Q77. Reverse a string

#include <iostream>

using namespace std;

int main(){
    string s;
    getline(cin,s);

    int i=0,j=s.length()-1;

        i++; j--;
    }

    cout << s;
}
