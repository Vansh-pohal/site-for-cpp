Q1. Write a C++ program to create an abstract class Animal with a sound() method. Derived classes: Lion & Tiger.

#include <iostream>
using namespace std;

class Animal {
public:
    virtual void sound() = 0;
};

class Lion : public Animal {
public:
    void sound() { cout << "Lion roars" << endl; }
};

class Tiger : public Animal {
public:
    void sound() { cout << "Tiger growls" << endl; }
};

int main() {
    Lion l;
    Tiger t;
    l.sound();
    t.sound();
    return 0;
}

Q3. Write a C++ program to create an abstract Shape class with calculateArea() and calculatePerimeter(). Derived: Circle & Triangle.

#include <iostream>
#include <cmath>
using namespace std;

class Shape {
public:
    virtual void calculateArea() = 0;
    virtual void calculatePerimeter() = 0;
};

class Circle : public Shape {
    double r;
public:
    Circle(double radius) { r = radius; }
    void calculateArea() { cout << 3.14*r*r << endl; }
    void calculatePerimeter() { cout << 2*3.14*r << endl; }
};

class Triangle : public Shape {
    double a,b,c;
public:
    Triangle(double x,double y,double z){a=x;b=y;c=z;}
    void calculateArea() {
        double s=(a+b+c)/2;
        cout << sqrt(s*(s-a)*(s-b)*(s-c)) << endl;
    }
    void calculatePerimeter(){ cout << a+b+c << endl;}
};

int main(){
    Circle c(5); c.calculateArea(); c.calculatePerimeter();
    Triangle t(3,4,5); t.calculateArea(); t.calculatePerimeter();
    return 0;
}

Q4. Write a C++ program to create an abstract BankAccount class with deposit() & withdraw(). Derived: SavingsAccount & CurrentAccount.

#include <iostream>
using namespace std;

class BankAccount{
public:
    virtual void deposit(double)=0;
    virtual void withdraw(double)=0;
};

class SavingsAccount: public BankAccount{
    double bal=0;
public:
    void deposit(double a){ bal+=a; }
    void withdraw(double a){ if(a<=bal) bal-=a; }
};

class CurrentAccount: public BankAccount{
    double bal=0;
public:
    void deposit(double a){ bal+=a; }
    void withdraw(double a){ bal-=a; }
};

int main(){
    SavingsAccount s; s.deposit(1000); s.withdraw(200);
    CurrentAccount c; c.deposit(2000); c.withdraw(500);
    return 0;
}
