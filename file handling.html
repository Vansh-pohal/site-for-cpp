Q8. Write a C++ program to sort the lines of a text file in alphabetical order using character array and getline.

#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int main() {
    ifstream fin("input.txt");
    ofstream fout("output.txt");

    if (!fin) {
        cout << "File not found!";
        return 0;
    }

    char lines[100][100];
    int count = 0;

    while (fin.getline(lines[count], 100)) {
        count++;
    }

    for (int i = 0; i < count - 1; i++) {
        for (int j = i + 1; j < count; j++) {
            if (strcmp(lines[i], lines[j]) > 0) {
                char temp[100];
                strcpy(temp, lines[i]);
                strcpy(lines[i], lines[j]);
                strcpy(lines[j], temp);
            }
        }
    }

    for (int i = 0; i < count; i++) {
        fout << lines[i] << endl;
    }

    cout << "Lines sorted successfully! Check output.txt";
}

OUTPUT

Lines sorted successfully! Check output.txt

Q9. Write a C++ program to merge multiple text files into a single file.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ofstream fout("merged.txt");

    if (!fout) {
        cout << "Error creating output file!";
        return 0;
    }

    int n;
    cout << "Enter number of files to merge: ";
    cin >> n;

    char filename[50];
    char ch;

    for (int i = 0; i < n; i++) {
        cout << "Enter file name " << i + 1 << ": ";
        cin >> filename;

        ifstream fin(filename);

        if (!fin) {
            cout << "Cannot open file: " << filename << endl;
            continue;
        }

        while (fin.get(ch)) {
            fout.put(ch);
        }

        fin.close();
    }

    cout << "Files merged successfully into merged.txt";

    fout.close();
    return 0;
}

OUTPUT

Enter number of files to merge: 2
Enter file name 1: file1.txt
Enter file name 2: file2.txt
Files merged successfully into merged.txt

Q10. Write a C++ program to split a large text file into smaller files of equal size.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream fin("input.txt");

    if (!fin) {
        cout << "File not found!";
        return 0;
    }

    int size;
    cout << "Enter number of characters per file: ";
    cin >> size;

    char ch;
    int count = 0, fileNo = 1;

    ofstream fout;
    string filename;

    filename = "output" + to_string(fileNo) + ".txt";
    fout.open(filename);

    while (fin.get(ch)) {
        fout.put(ch);
        count++;

        if (count == size) {
            fout.close();
            fileNo++;
            filename = "output" + to_string(fileNo) + ".txt";
            fout.open(filename);
            count = 0;
        }
    }

    fin.close();
    fout.close();

    cout << "File split successfully!";

    return 0;
}

OUTPUT

Enter number of characters per file: 50
File split successfully!

Q11. Write a C++ program to search for a specific string in a text file and display its line number(s).

#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int main() {
    ifstream fin("input.txt");

    if (!fin) {
        cout << "File not found!";
        return 0;
    }

    char line[200];
    char search[50];
    int lineNo = 0;
    bool found = false;

    cout << "Enter string to search: ";
    cin.getline(search, 50);

    while (fin.getline(line, 200)) {
        lineNo++;

        if (strstr(line, search) != NULL) {
            cout << "Found at line number: " << lineNo << endl;
            found = true;
        }
    }

    if (!found) {
        cout << "String not found!";
    }

    fin.close();
    return 0;
}

OUTPUT

Enter string to search: HELLO
Found at line number: 1

Q16. Write a C++ program to calculate the average of numbers stored in a file.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream fin("numbers.txt");

    int num, sum = 0, count = 0;

    while (fin >> num) {
        sum += num;
        count++;
    }

    if (count > 0)
        cout << "Average = " << (float)sum / count;
    else
        cout << "No data";

    fin.close();
    return 0;
}

Q17. Write a C++ program to store and manage student records using a binary file.

#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

class Student {
    int rollno;
    char name[50];
    float marks;
    char mobileno[15];

public:
    void get() {
        cin >> rollno >> name >> marks >> mobileno;
    }

    void show() {
        cout << rollno << "\t" << name << "\t" << marks << "\t" << mobileno << endl;
    }

    int getRoll() {
        return rollno;
    }
};

int main() {
    Student s;
    fstream file, temp;
    int ch, r;
    bool found;

    do {
        cout << "\n1.Insert 2.Delete 3.Update 4.Search 5.Clone 6.Exit\n";
        cin >> ch;

        switch (ch) {

        case 1:
            file.open("student.dat", ios::binary | ios::app);
            s.get();
            file.write((char*)&s, sizeof(s));
            file.close();
            break;

        case 2:
            cin >> r;
            file.open("student.dat", ios::binary | ios::in);
            temp.open("temp.dat", ios::binary | ios::out);

            while (file.read((char*)&s, sizeof(s))) {
                if (s.getRoll() != r)
                    temp.write((char*)&s, sizeof(s));
            }

            file.close();
            temp.close();
            remove("student.dat");
            rename("temp.dat", "student.dat");
            break;

        case 3:
            cin >> r;
            file.open("student.dat", ios::binary | ios::in | ios::out);

            while (file.read((char*)&s, sizeof(s))) {
                if (s.getRoll() == r) {
                    s.get();
                    file.seekp(-sizeof(s), ios::cur);
                    file.write((char*)&s, sizeof(s));
                }
            }

            file.close();
            break;

        case 4:
            cin >> r;
            file.open("student.dat", ios::binary | ios::in);
            found = false;

            while (file.read((char*)&s, sizeof(s))) {
                if (s.getRoll() == r) {
                    s.show();
                    found = true;
                }
            }

            if (!found) cout << "Not found";

            file.close();
            break;

        case 5:
            file.open("student.dat", ios::binary | ios::in);
            temp.open("clone.dat", ios::binary | ios::out);

            while (file.read((char*)&s, sizeof(s))) {
                temp.write((char*)&s, sizeof(s));
            }

            file.close();
            temp.close();
            break;
        }

    } while (ch != 6);

    return 0;
}

OUTPUT

1.Insert 2.Delete 3.Update 4.Search 5.Clone 6.Exit
Record inserted / searched / updated successfully
