Q1. Program to Check Anagram

#include <iostream>
#include <string>
using namespace std;

bool isAnagram(string first, string second){
    if (first.length() != second.length()) return false;
    int freq[256] = {0};
    for (int i = 0; i < (int)first.length(); i++){
        freq[(int)first[i]]++;
        freq[(int)second[i]]--;
    }
    for (int i = 0; i < 256; i++){
        if (freq[i] != 0) return false;
    }
    return true;
}

int main(){
    string first, second;
    cout << "Enter first string: ";
    cin >> first;
    cout << "Enter second string: ";
    cin >> second;
    if (isAnagram(first, second)) cout << "Anagram\n";
    else cout << "Not anagram\n";
    return 0;
}

OUTPUT

Enter first string: listen
Enter second string: silent
Anagram

Q14. Program to determine whether one string is a rotation of another

#include <iostream>
#include <string>
using namespace std;

bool containsText(string text, string pattern){
    if (pattern.length() > text.length()) return false;
    for (int i = 0; i <= (int)text.length() - (int)pattern.length(); i++){
        int match = 1;
        for (int j = 0; j < (int)pattern.length(); j++){
            if (text[i + j] != pattern[j]){ match = 0; break; }
        }
        if (match == 1) return true;
    }
    return false;
}

int main(){
    string first, second;
    cout << "Enter first string: ";
    cin >> first;
    cout << "Enter second string: ";
    cin >> second;
    if (first.length() != second.length()){ cout << "Not rotation\n"; return 0; }
    string doubled = first + first;
    if (containsText(doubled, second)) cout << "Rotation\n";
    else cout << "Not rotation\n";
    return 0;
}

OUTPUT

Enter first string: ABCD
Enter second string: CDAB
Rotation

Q17. Program to find all the permutations of a string

#include <iostream>
#include <string>
using namespace std;

void swapChar(char &a, char &b){ char temp=a;a=b;b=temp; }

void printPermutations(string &text,int start,int end){
    if(start==end){ cout << text << "\n"; return; }
    for(int i=start;i<=end;i++){
        swapChar(text[start],text[i]);
        printPermutations(text,start+1,end);
        swapChar(text[start],text[i]);
    }
}

int main(){
    string text;
    cout << "Enter a string: ";
    cin >> text;
    cout << "Permutations are:\n";
    printPermutations(text,0,(int)text.length()-1);
    return 0;
}

Q18. Program to find maximum and minimum occurring character in a string

    getline(cin,text);
    if(text.length()==0){ cout << "Empty string\n"; return 0; }
    for(int i=0;i<(int)text.length();i++){ if(text[i]!=' ') freq[(int)text[i]]++; }
    int maxCount=0,minCount=0; char maxChar=' ',minChar=' ';
    for(int i=0;i<256;i++){ if(freq[i]>0){ maxCount=freq[i]; minCount=freq[i]; maxChar=(char)i; minChar=(char)i; break; } }
    for(int i=0;i<256;i++){
        if(freq[i]>0){
            if(freq[i]>maxCount){ maxCount=freq[i]; maxChar=(char)i; }
            if(freq[i]<minCount){ minCount=freq[i]; minChar=(char)i; }
        }
    }
    cout << "Maximum occurring character: " << maxChar << "\n";
    cout << "Minimum occurring character: " << minChar << "\n";
    return 0;
}

Q24. Program to find the longest repeating sequence in a string

#include <iostream>
#include <string>
using namespace std;

int main(){
    string text;
    string longest="";
    cout << "Enter a string: ";
    cin >> text;
    int n = (int)text.length();

    for(int len=1; len<=n/2; len++){
        for(int i=0; i+2*len <= n; i++){
            int same=1;
            for(int j=0; j<len; j++){
                if(text[i+j] != text[i+len+j]){
                    same=0;
                    break;
                }
            }
            if(same==1 && len > (int)longest.length()){
                longest = text.substr(i,len);
            }
        }
    }

    if(longest.length()==0){
        cout << "No repeating sequence found\n";
    } else {
        cout << "Longest repeating sequence: " << longest << "\n";
    }
    return 0;
}

Q25. Program to find the most repeated word in a text file

#include <iostream>
#include <string>
#include <fstream>
using namespace std;

int main(){
    string fileName;
    string words[1000];
    int wordCount=0;

    cout << "Enter file name: ";
    cin >> fileName;

    ifstream file(fileName);
    if(!file){
        cout << "Unable to open file\n";
        return 0;
    }

    string word;
    while(file >> word){
        words[wordCount]=word;
        wordCount++;
        if(wordCount >= 1000) break;
    }
    file.close();

    if(wordCount==0){
        cout << "No words in file\n";
        return 0;
    }

    int maxCount=0;
    string maxWord="";

    for(int i=0;i<wordCount;i++){
        int count=0;
        for(int j=0;j<wordCount;j++){
            if(words[i]==words[j]) count++;
        }
        if(count>maxCount){
            maxCount=count;
            maxWord=words[i];
        }
    }

    cout << "Most repeated word: " << maxWord << "\n";
    cout << "Frequency: " << maxCount << "\n";
    return 0;
}

}

Q48. Write a C++ method to check whether a string is a valid password

#include <iostream>
#include <string>
using namespace std;

bool validPassword(string password){
    if((int)password.length()<8) return false;
    int hasUpper=0, hasLower=0, hasDigit=0, hasSpecial=0;
    for(int i=0;i<(int)password.length();i++){
        char ch=password[i];
        if(ch>='A' && ch<='Z') hasUpper=1;
        else if(ch>='a' && ch<='z') hasLower=1;
        else if(ch>='0' && ch<='9') hasDigit=1;
        else hasSpecial=1;
    }
    return hasUpper && hasLower && hasDigit && hasSpecial;
}

int main(){
    string password;
    cout << "Enter password: "; cin >> password;
    if(validPassword(password)) cout << "Valid password\n";
    else cout << "Invalid password\n";
    return 0;
}
